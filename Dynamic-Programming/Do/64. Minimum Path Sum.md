// 64. Minimum Path Sum  
https://leetcode.com/problems/minimum-path-sum/  
```
class Solution {
public:
  //状态转移方程：dp[i][j]表示为到dp[i][j]点最多有多少条路=dp[i - 1][j] + dp[i][j - 1]
  int minPathSum(vector<vector<int>> &grid) {
    int m = grid.size();
    int n = grid[0].size();
    vector<vector<int>> dp = grid;
    for (int j = 1; j < n; j++)
      dp[0][j] += dp[0][j - 1];

    for (int i = 1; i < m; i++)
      dp[i][0] += dp[i - 1][0];
    
    for (int i = 1; i < m; i++) {
      for (int j = 1; j < n; j++) {
        dp[i][j] += min(dp[i - 1][j], dp[i][j - 1]);
      }
    }
    return dp[m - 1][n - 1];
  }
};
```
