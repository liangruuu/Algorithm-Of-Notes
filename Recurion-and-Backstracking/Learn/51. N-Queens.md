// 51. N-Queens  
https://leetcode.com/problems/n-queens/
```
          解决n皇后问题
      /    |     |    \
     0     1     2     3   row1
    / \
   2   3                   row2
(0,1因为条件不符被剪枝了)


快速判断不合法的情况
1. 竖向：col[i]=true表示第i列被占用
2. 对角线1：dia1[i]表示第i对角线1被占用
3. 对角线2：dia2[i]表示第i对角线2被占用
4. 不需要判断行不和法,因为每一行必须得放一个元素

00 01 02 03
10 11 12 13   
20 21 22 23   
30 31 32 33
1. 对角线1(右上到左下)：副对角线(2*n - 1个), 每一个对角线上的元素相加的和为一定值(1,2,3,4....i+j)
2. 对角线2(左上到右下)：主对角线(2*n - 1个), 每一个对角线上的元素相减的和为一定值(-3,-2,-1,0....i-j), 这里需要注意的是数组下标是从0开始的,所以我们需要对数值进行处理,即把i-j改为i-j+n-1(-3变为了0, -2变为了1...)
```
```
/// 时间复杂度: O(n^n)
/// 空间复杂度: O(n)
class Solution {
private:
  vector<bool> col, dia1, dia2;
  vector<vector<string>> res;

  // 尝试在一个n皇后问题中, 摆放第index行的皇后位置
  void putQueen(int n, int index, vector<int> &row) {

    if (index == n) {
      res.push_back(generateBoard(n, row));
      return;
    }

    for (int i = 0; i < n; i++)
      // 尝试将第index行的皇后摆放在第i列
      if (!col[i] && !dia1[index + i] && !dia2[index - i + n - 1]) {
        row.push_back(i);
        col[i] = true;
        dia1[index + i] = true;
        dia2[index - i + n - 1] = true;
        putQueen(n, index + 1, row);
        col[i] = false;
        dia1[index + i] = false;
        dia2[index - i + n - 1] = false;
        row.pop_back();
      }

    return;
  }

  vector<string> generateBoard(int n, vector<int> &row) {
    assert(row.size() == n);
    vector<string> board(n, string(n, '.'));
    for (int i = 0; i < n; i++)
      board[i][row[i]] = 'Q';
    return board;
  }

public:
  vector<vector<string>> solveNQueens(int n) {

    res.clear();

    col.clear();
    for (int i = 0; i < n; i++)
      col.push_back(false);
    // 等价于col = vector<bool>(n, false);

    dia1.clear();
    dia2.clear();
    for (int i = 0; i < 2 * n - 1; i++) {
      dia1.push_back(false);
      dia2.push_back(false);
    }

    vector<int> row;
    putQueen(n, 0, row);

    return res;
  }
};
```
